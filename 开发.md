## 1. 项目概述

**项目名称**: Arknights Auto Machine (AAM)  
**版本**: 0.1.0-alpha  
**架构**: AMA (Associative Motor Architecture) L0-L5 分层控制  
**运行环境**: Windows/Linux + Android模拟器 (ADB)  
**技术栈**: Python 3.10+, ttkbootstrap (GUI), adb-shell, OpenCV/PIL, LiteLLM/自定义API客户端

---

## 2. 架构设计 (AMA Implementation)

### 2.1 分层实现映射

| AMA层级 | 模块名 | 职责 | 技术实现 |
|---------|--------|------|----------|
| **L5** | `aam.brain` | 视觉理解/策略决策 | 连接Qwen-VL/GPT-4V等VLM API，输出标准化JSON指令 |
| **L4** | `aam.memory` | 对局状态机/存档 | SQLite + Pydantic模型，保存关卡进度、干员状态 |
| **L3** | `aam.predictor` | 敌人轨迹预测 | OpenCV光流法 + 简单线性外推，预测敌人3秒后的位置 |
| **L2** | `aam.mapper` | 语义→空间转换 | 坐标映射表(关卡JSON配置)，将"第3格"转换为(650,650)像素 |
| **L1** | `aam.reflex` | 快速反射/应急 | 独立线程，监听血线危急信号，直接触发撤退指令 |
| **L0** | `aam.sensor` + `aam.motor` | 截图输入/触控输出 | `adb shell screencap` + `adb shell input tap/swipe` |

### 3.1 L0-Sensor (截图模块)
```python
# aam/sensor.py
class AdbSensor:
    def __init__(self, device_serial: str = None):
        self.device = adbclient.device(device_serial) if device_serial else adbclient
    
    def capture(self) -> np.ndarray:
        """使用adb截图并转为cv2格式"""
        raw = self.device.shell("screencap -p", decode=False)
        # 处理PNG二进制流
        img = np.frombuffer(raw, dtype=np.uint8)
        return cv2.imdecode(img, cv2.IMREAD_COLOR)
    
    def stream(self, fps: int = 10):
        """生成器模式，持续截图"""
        while True:
            yield self.capture()
            time.sleep(1/fps)
```

### 3.2 L0-Motor (执行模块)
```python
# aam/motor.py
class AdbMotor:
    def tap(self, x: int, y: int, duration: int = 50):
        """点击，duration为按压时长(ms)"""
        self.device.shell(f"input tap {x} {y}")
    
    def swipe(self, x1, y1, x2, y2, duration=200):
        """滑动，用于拖放干员"""
        self.device.shell(f"input swipe {x1} {y1} {x2} {y2} {duration}")
    
    def deploy_operator(self, slot_idx: int, grid_x: int, grid_y: int, direction: str):
        """L2层调用的原子操作：部署干员"""
        # 从干员栏拖拽到地图
        start_pos = self._get_operator_slot_pos(slot_idx)  # L2提供映射
        end_pos = self._get_grid_pos(grid_x, grid_y)
        self.swipe(*start_pos, *end_pos, duration=200)
        time.sleep(0.1)
        # 方向调整（微拖）
        self._adjust_direction(end_pos, direction)
```

### 4.1 关卡配置JSON格式
```json
{
  "level_id": "1-7",
  "resolution": [1920, 1080],
  "grid_system": {
    "cols": 9,
    "rows": 5,
    "origin": [300, 200],
    "cell_size": [120, 120]
  },
  "ui_elements": {
    "operator_bar": {"y_range": [950, 1050], "slots": 12},
    "speed_button": [1800, 80],
    "pause_button": [100, 80]
  },

    "grid_to_pixel": "lambda x,y: (300+x*120, 200+y*120)"
  }

```

### 4.2 L2映射引擎
```python
# aam/mapper.py
class CoordinateMapper:
    def __init__(self, level_config_path: str):
        self.config = json.load(open(level_config_path))
        self.resolution = self.config['resolution']
    
    def grid_to_pixel(self, col: int, row: int) -> Tuple[int, int]:
        """L2核心：网格坐标转屏幕像素"""
        origin = self.config['grid_system']['origin']
        cell = self.config['grid_system']['cell_size']
        return (
            origin[0] + col * cell[0] + cell[0]//2,
            origin[1] + row * cell[1] + cell[1]//2
        )
    
    def operator_slot_to_pixel(self, slot_idx: int) -> Tuple[int, int]:
        """干员栏第n个位置"""
        bar = self.config['ui_elements']['operator_bar']
        x_start = 100  # 根据实际UI测量
        spacing = (1920 - 200) // bar['slots']
        return (x_start + slot_idx * spacing, (bar['y_range'][0] + bar['y_range'][1])//2)
    
    def execute_l5_command(self, command: dict):
        """执行L5发来的JSON指令"""
        if command['action'] == 'deploy':
            slot = command['operator_index']
            pos = command['position']  # [col, row]
            direction = command['direction']
            
            # 转换为L0像素坐标
            start = self.operator_slot_to_pixel(slot)
            end = self.grid_to_pixel(*pos)
            
            # 调用L0执行
            self.motor.swipe(*start, *end)
            self.motor.adjust_direction(end, direction)
```

```python
# aam/brain/vision.py
class VisionEngine:
    def __init__(self, api_key: str, model: str = "qwen-vl-max"):
        self.client = openai.OpenAI(api_key=api_key, base_url="...")
        self.model = model
    
    def analyze_screen(self, img: np.ndarray) -> dict:
        """分析当前画面，返回结构化指令"""
        # 压缩图片base64编码
        _, buffer = cv2.imencode('.jpg', img, [int(cv2.IMWRITE_JPEG_QUALITY), 85])
        img_b64 = base64.b64encode(buffer).decode()
        
        prompt = """你是明日方舟战术AI。分析当前战场：
        1. 可用干员（底部栏位）
        2. 敌人位置与行进路线（红框标记）
        3. 费用情况（左上角COST）
        
        返回JSON格式决策：
        {
            "action": "deploy" | "skill" | "wait",
            "operator_index": int,
            "target_position": [col, row],
            "direction": "left" | "right" | "up" | "down",
            "confidence": float,
            "reasoning": str
        }"""
        
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[{
                "role": "user",
                "content": [
                    {"type": "text", "text": prompt},
                    {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{img_b64}"}}
                ]
            }]
        )
        
        return json.loads(response.choices[0].message.content)
```

**目标**: 减少LLM调用频率，提升响应速度

### 6.1 L3运动预测（轻量级）
```python
# aam/predictor/trajectory.py
class EnemyPredictor:
    """基于传统CV的轻量预测，不调用LLM"""
    def __init__(self):
        self.tracks = {}  # enemy_id -> [(x,y,t), ...]
        self.kalman_filters = {}
    
    def update_tracks(self, detections: List[Tuple[int, int]]):
        """接收L5或本地CV检测的敌人位置"""
        # 简化的线性预测：基于最近3帧计算速度向量
        for enemy_id, pos in detections:
            if enemy_id not in self.tracks:
                self.tracks[enemy_id] = deque(maxlen=5)
            self.tracks[enemy_id].append((pos, time.time()))
    
    def predict_position(self, enemy_id: str, seconds_ahead: float = 1.0) -> Tuple[int, int]:
        """预测n秒后的位置，用于提前部署"""
        track = self.tracks.get(enemy_id)
        if len(track) < 2:
            return track[-1][0] if track else (0,0)
        
        # 线性外推
        (x1, y1), t1 = track[-2]
        (x2, y2), t2 = track[-1]
        vx = (x2 - x1) / (t2 - t1)
        vy = (y2 - y1) / (t2 - t1)
        
        return (int(x2 + vx * seconds_ahead), int(y2 + vy * seconds_ahead))
```

### 6.2 L1反射弧（紧急响应）
```python
# aam/reflex/emergency.py
class ReflexArc:
    """独立线程，高频检测紧急情况"""
    def __init__(self, sensor: AdbSensor, motor: AdbMotor):
        self.sensor = sensor
        self.motor = motor
        self.running = False
        self.emergency_threshold = 0.3  # 血线30%触发
    
    def start_monitoring(self):
        """启动守护线程"""
        self.running = True
        Thread(target=self._monitor_loop, daemon=True).start()
    
    def _monitor_loop(self):
        while self.running:
            img = self.sensor.capture()
            danger_zones = self._detect_danger(img)  # 快速模板匹配
            
            for zone in danger_zones:
                if zone['hp_ratio'] < self.emergency_threshold:
                    # 直接触发L0撤退，不经过L5
                    self.motor.trigger_skill(zone['operator_pos'])  # 开技能保命
                    time.sleep(0.5)  # 冷却
```

## 8. 风险与限制

1. **ADB依赖**: 仅支持模拟器/ADB调试手机，不支持iOS
2. **分辨率敏感**: 需要为每种分辨率(1080p/2k)单独制作L2映射
3. **VLM成本**: L5层调用费用较高，需配合L3缓存优化
4. **反检测**: 需要控制操作间隔随机化，避免被识别为脚本
